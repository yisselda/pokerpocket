React Component 
import { useState } from 'react'
import { newGame, betting } from 'pokerpocket'

function PokerGame() {
  const [gameState, setGameState] = useState(null)
  const [chipStacks, setChipStacks] = useState([])
  const [pot, setPot] = useState(0)

  const startNewGame = () => {
    const game = newGame({ players: 4 })
    let bettingState = betting.initBettingWithDefaults(4, 0, 10000)

    // Deal and post blinds
    game.deal()
    bettingState = betting.postBlinds(bettingState, {
      smallBlind: 50, bigBlind: 100
    })

    // Play through streets
    game.flop()
    game.turn()
    game.river()

    // Get results and distribute pot
    const result = game.showdown()
    const distribution = betting.distributePots(bettingState, result.winners)

    // Update final stacks
    const finalStacks = bettingState.players.map((p, i) => {
      const winnings = distribution.find(d => d.player === i)?.amount || 0
      return p.stack + winnings
    })

    setGameState(game.status())
    setChipStacks(finalStacks)
    setPot(betting.getTotalPot(bettingState))
  }

  return (
    <div>
      <button onClick={startNewGame}>Deal New Hand</button>
      {gameState && (
        <div>
          <p>Board: {gameState.boardAscii}</p>
          <p>Pot: {pot} chips</p>
          <p>Chip Stacks: {chipStacks.join(', ')}</p>
        </div>
      )}
    </div>
  )
}
Node.js Server 
import express from 'express'
import { newGame, betting } from 'pokerpocket'

const app = express()
app.use(express.json())

// Tournament endpoint with betting
app.post('/api/poker/simulate', (req, res) => {
  const { players = 6, rounds = 100, buyIn = 10000 } = req.body
  const results = []
  const chipCounts = new Array(players).fill(buyIn)

  for (let i = 0; i < rounds; i++) {
    const game = newGame({ players, seed: Date.now() + i })
    let bettingState = betting.initBetting(players, chipCounts, i % players)

    // Deal and post blinds
    game.deal()
    bettingState = betting.postBlinds(bettingState, {
      smallBlind: 50, bigBlind: 100
    })

    // Simulate betting (simplified)
    game.flop()
    game.turn()
    game.river()

    // Distribute winnings
    const result = game.showdown()
    const distribution = betting.distributePots(bettingState, result.winners)

    // Update chip counts
    distribution.forEach(({ player, amount }) => {
      chipCounts[player] += amount
    })

    results.push({
      round: i + 1,
      winners: result.winners,
      pot: betting.getTotalPot(bettingState),
      chipCounts: [...chipCounts]
    })
  }

  res.json({ tournament: results })
})

app.listen(3000, () => {
  console.log('Poker tournament server with betting on port 3000')
})
TypeScript with Full Types 
import {
  newGame,
  betting,
  type GameResult,
  type BettingState,
  type Action
} from 'pokerpocket'

interface PokerSession {
  id: string
  players: number
  chipStacks: number[]
  handsPlayed: number
}

class PokerTournament {
  private sessions: Map<string, PokerSession> = new Map()
  private readonly defaultBuyIn = 10000
  private readonly blinds = { smallBlind: 50, bigBlind: 100 }

  createTournament(players: number): string {
    const id = crypto.randomUUID()
    this.sessions.set(id, {
      id,
      players,
      chipStacks: new Array(players).fill(this.defaultBuyIn),
      handsPlayed: 0
    })
    return id
  }

  playHand(sessionId: string): {
    winners: number[]
    pot: number
    chipStacks: number[]
  } {
    const session = this.sessions.get(sessionId)
    if (!session) throw new Error('Session not found')

    // Create game and betting state
    const game = newGame({
      players: session.players,
      seed: Date.now() + session.handsPlayed
    })

    let bettingState = betting.initBetting(
      session.players,
      session.chipStacks,
      session.handsPlayed % session.players
    )

    // Deal and post blinds
    game.deal()
    bettingState = betting.postBlinds(bettingState, this.blinds)

    // Simulate simple betting
    game.flop()
    game.turn()
    game.river()

    // Showdown and distribute
    const result = game.showdown()
    const distribution = betting.distributePots(bettingState, result.winners)
    const pot = betting.getTotalPot(bettingState)

    // Update chip stacks
    distribution.forEach(({ player, amount }) => {
      session.chipStacks[player] += amount
    })

    session.handsPlayed++

    return {
      winners: result.winners,
      pot,
      chipStacks: [...session.chipStacks]
    }
  }

  getChipLeader(sessionId: string): number | null {
    const session = this.sessions.get(sessionId)
    if (!session) return null

    const maxChips = Math.max(...session.chipStacks)
    return session.chipStacks.indexOf(maxChips)
  }
}

CLI Commands

deal Deal new hand with blinds posted
flop Deal flop (3 community cards)
turn Deal turn (4th community card)
river Deal river (5th community card)
showdown Evaluate hands and distribute pot
check Check (when no bet to call)
call Call the current bet
bet <amount> Bet specified amount
raise <amount> Raise by specified amount
fold Fold your hand
allin Go all-in with remaining chips
blinds <sb> <bb> Set small and big blind amounts
ante <amount> Set ante amount (0 for none)
stacks <amount> Set default starting stack size
button <player> Set dealer button position (0-based)
players <n> Set number of players (2-9)
seed <n> Set RNG seed for reproducible games
skipbet Auto-complete betting round (testing)
status Show current game state
help Show all available commands

API Reference

Complete reference for PokerPocket engine methods

Engine Methods

newGame(options?: GameOptions): PokerEngine

Create a new poker game instance with optional configuration.

Example: 
const game = newGame({ players: 6, seed: 12345 })
game.setPlayers(count: number): void

Set number of players (2-9). Only available in IDLE phase.

Example: 
game.setPlayers(4)
game.setSeed(seed: number): void

Set RNG seed for next deal to ensure reproducible games.

Example: 
game.setSeed(42)
game.deal(): void

Deal hole cards (2 cards per player) and start new hand.

Example: 
game.deal()
game.flop(): void

Deal the flop (3 community cards). Burns 1 card first.

Example: 
game.flop()
game.turn(): void

Deal the turn (4th community card). Burns 1 card first.

Example: 
game.turn()
game.river(): void

Deal the river (5th community card). Burns 1 card first.

Example: 
game.river()
game.showdown(): ShowdownResult

Evaluate all hands and determine winners. Returns results and winner indices.

Example: 
const { results, winners } = game.showdown()
game.status(): GameStatus

Get current game state including phase, players, and cards.

Example: 
const status = game.status()

Evaluation Methods

evaluate7(cards: Card[]): EvalResult

Evaluate 7 cards and return best 5-card hand with score and rank.

Example: 
const result = evaluate7([...holeCards, ...board])
console.log(result.rank) // 'FULL_HOUSE'

Betting Methods

betting.initBetting(playerCount: number, stacks: number[], buttonIndex: number): BettingState

Initialize betting state with custom stack sizes for each player.

Example: 
const state = betting.initBetting(3, [1000, 1500, 800], 0)
betting.initBettingWithDefaults(playerCount: number, buttonIndex: number, defaultStack?: number): BettingState

Initialize betting state with equal stacks for all players (default 10000).

Example: 
const state = betting.initBettingWithDefaults(4, 1, 5000)
betting.postBlinds(state: BettingState, config: TableConfig): BettingState

Post blinds and antes according to table configuration. Returns updated state.

Example: 
state = betting.postBlinds(state, { smallBlind: 25, bigBlind: 50 })
betting.legalActions(state: BettingState, player: number): ActionInfo[]

Get all legal actions for a player with min/max amounts where applicable.

Example: 
const actions = betting.legalActions(state, 2)
// [{ type: 'fold' }, { type: 'call' }, { type: 'raise', min: 100, max: 500 }]
betting.applyAction(state: BettingState, action: Action): BettingState

Apply a betting action and return the updated state. Validates action legality.

Example: 
state = betting.applyAction(state, { player: 1, type: 'raise', amount: 200 })
betting.isRoundComplete(state: BettingState): boolean

Check if the current betting round is complete (all players acted or folded).

Example: 
if (betting.isRoundComplete(state)) {
  // Advance to next street
}
betting.startNewRound(state: BettingState): BettingState

Start a new betting round (flop/turn/river). Builds pots and resets for new round.

Example: 
game.flop()
state = betting.startNewRound(state)
betting.distributePots(state: BettingState, winners: number[]): Distribution[]

Distribute all pots to winners, handling side pots automatically.

Example: 
const { winners } = game.showdown()
const winnings = betting.distributePots(state, winners)
betting.getTotalPot(state: BettingState): number

Get the total amount of chips in play (all committed chips).

Example: 
const pot = betting.getTotalPot(state) // 450
betting.getActivePlayers(state: BettingState): number[]

Get array of player indices who haven't folded.

Example: 
const active = betting.getActivePlayers(state) // [0, 2, 3]

Utility Methods

drawRandom(count: number, seed?: number): Card[]

Generate random cards for testing or demos. Optionally seeded.

Example: 
const randomHand = drawRandom(7, 42)
cardToAscii(card: Card): string

Convert a card object to ASCII representation with Unicode suit symbols.

Example: 
import { cardToAscii } from 'pokerpocket'
const card = { rank: 'A', suit: 's' }
console.log(cardToAscii(card)) // 'A‚ô†'

Basic card representation with rank ('2'-'9', 'T', 'J', 'Q', 'K', 'A') and suit ('s', 'h', 'd', 'c').

Example: 
// state.currentBet = 100

Betting action with player index, action type, and optional amount for bets/raises.

TableConfig = { smallBlind: number; bigBlind: number; ante?: number }

Table configuration for blinds and antes.
